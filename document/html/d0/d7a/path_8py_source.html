00013 def walk(top, maxDepth=None, onerror=None, followlinks=False):
00014         """
00015         @brief 재귀적으로 경로 탐색하기
00016         @details '.'를 인자로 주었다면, './a', './b' 순으로 점점 더 깊게 탐색합니다.
00017         @param top 탐색 대상 경로
00018         @param onerror 오류가 발생하면, 주어진 주소의 함수를 호출하여 오류 인자를 넘겨줍니다.
00019         @param followlinks 심볼릭 링크된 경로를 포함하여 탐색합니다.
00020         @param maxDepth 탐색 할 최대 깊이\n
00021                 0이상의 정수여야 함.\n
00022                 0 : 지정된 폴더만
00023         """
00024         top = top.rstrip(os.path.sep)
00025         assert os.path.isdir(top)
00026         num_sep = top.count(os.path.sep)
00027         for root, dirs, files in os.walk(top, onerror=onerror, followlinks=followlinks):
00028                 yield root, dirs, files
00029                 if maxDepth is not None and num_sep + maxDepth <= root.count(os.path.sep):
00030                         del dirs[:]
00031                 pass
00032         pass

00122                                 elif options['patternReverse']:
00123                                         listdir.append(filename)
00124                                         break
00125                                 pass
00126                         pass

00128         
00129         #sortKey 옵션이 사용된 경우
00130         if not options['sortKey']:
00131                 buffer = options['sortKey']

00224 def unique(targetPath, spliteExt = True) :
00225         """
00226         @brief 해당 경로에서 유일한 경로를 만들어 냅니다.
00227         @details \n
00228                 대소문자는 구분하지 않습니다.\n
00229                 만약, 주어진 경로가 충돌하지 않는다면 주어진 경로를 그대로 반환합니다.\n
00230                 만약 주어진 경로의 부모 경로가 존재하지 않는다면, 주어진 문자열 그대로 반환합니다.\n
00231                 ex) a/b/c -> a/b/c d(1)식으로 충복을 회피처리함.
00232         @param targetPath 주어지는 값은 유니코드 문자열이여야 함.

00238         targetDirname = dirname(targetPath);
00239         targetBasename = os.path.basename(targetPath);
00240

00242         if not os.path.exists(targetDirname) :
00243                 return targetPath;

00245         #해당 경로의 목록을 작성
00246         fileList = [u'.', u'..'];
00247         for dirpath, dirnames, filenames in os.walk(targetDirname) :
00248                 fileList.extend(filenames);
00249                 break;
00250                 
00251         #중복되는 대상이 존재하는지 확인
00252         existDuplicateFile = False;
00253         buffer = re.escape(targetBasename);
00254         fullmatchRe = re.compile(u"^%(buffer)s$" % locals(), re.DOTALL | re.IGNORECASE | re.UNICODE);
00255         for fileName in fileList :
00256                 if fullmatchRe.search(fileName) :
00257                         existDuplicateFile = True;
00258                         break;
00259         
00260         #중복되는 대상이 존재하고 있지 않다면 이름을 그대로 돌려줌.
00261         if not existDuplicateFile :
00262                 return targetPath;

00264         #파일명과 확장자를 추출
00265         if spliteExt :

00268                         targetFileName = filename(targetBasename);
00269                 else :

00277         if spliteExt :
00278                 extractDupCountRe = re.compile(ur"^%(escapedTargetFileName)s \(d(?P<number>[0-9]+)\)\.%(targetFileExt)s$" % locals(), re.DOTALL | re.IGNORECASE | re.UNICODE);

00280                 extractDupCountRe = re.compile(ur"^%(escapedTargetFileName)s \(d(?P<number>[0-9]+)\)$" % locals(), re.DOTALL | re.IGNORECASE | re.UNICODE);
00281         counts = [];

00304 def filename(filePath) :
00305         """
00306         @brief 파일 경로로 부터 확장자를 제외한 파일명을 추출해냅니다.

00310         rx = re.compile(ur"^(.*/)?(?P<name_space>.+?)?(?P<ext_space>\.[a-z0-9]+)?$", re.DOTALL | re.IGNORECASE | re.UNICODE);
00311         
00312         result = rx.search(filePath)
00313         if not result:
00314                 return unicode()
00315                 
00316         nameSpace = result.group('name_space');
00317         if nameSpace :
00318                 return nameSpace;
00319         
00320         extSpace = result.group('ext_space')
00321         if extSpace:
00322                 return extSpace;
00323                 
00324         return unicode()
